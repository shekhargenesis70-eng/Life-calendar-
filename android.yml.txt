          package com.lifecalendar;
          import android.content.SharedPreferences;
          import android.graphics.Canvas;
          import android.graphics.Color;
          import android.graphics.Paint;
          import android.graphics.RectF;
          import android.os.Handler;
          import android.os.Looper;
          import android.service.wallpaper.WallpaperService;
          import android.view.SurfaceHolder;
          import java.util.Calendar;
          import java.util.TimeZone;
          public class LifeCalendarWallpaperService extends WallpaperService {
              @Override public Engine onCreateEngine(){return new CalendarEngine();}
              class CalendarEngine extends Engine {
                  private final Handler handler=new Handler(Looper.getMainLooper());
                  private final Runnable ticker=this::draw;
                  private boolean visible=true;
                  private float pulse=1f,pulseDir=0.02f;
                  private Paint bg,done,today,left,clock,date,stats,barBg,bar;
                  @Override public void onCreate(SurfaceHolder s){
                      super.onCreate(s);
                      bg=p("#0a0a0a");done=p("#ffffff");today=p("#ff4f6d");left=p("#2a2a2a");
                      clock=p("#ffffff");clock.setFakeBoldText(true);clock.setTextAlign(Paint.Align.CENTER);
                      date=p("#666666");date.setTextAlign(Paint.Align.CENTER);
                      stats=p("#ff4f6d");stats.setTextAlign(Paint.Align.CENTER);
                      barBg=p("#1e1e1e");bar=p("#ff4f6d");
                  }
                  Paint p(String c){Paint x=new Paint(Paint.ANTI_ALIAS_FLAG);x.setColor(Color.parseColor(c));return x;}
                  @Override public void onVisibilityChanged(boolean v){visible=v;if(v)handler.post(ticker);else handler.removeCallbacks(ticker);}
                  @Override public void onSurfaceDestroyed(SurfaceHolder h){visible=false;handler.removeCallbacks(ticker);}
                  @Override public void onSurfaceChanged(SurfaceHolder h,int f,int w,int ht){draw();}
                  void draw(){
                      SurfaceHolder h=getSurfaceHolder();Canvas c=null;
                      try{c=h.lockCanvas();if(c!=null)render(c);}finally{if(c!=null)h.unlockCanvasAndPost(c);}
                      handler.removeCallbacks(ticker);
                      if(visible){pulse+=pulseDir;if(pulse>1.15f)pulseDir=-0.02f;if(pulse<1f)pulseDir=0.02f;handler.postDelayed(ticker,50);}
                  }
                  void render(Canvas c){
                      float W=c.getWidth(),H=c.getHeight();
                      c.drawRect(0,0,W,H,bg);
                      SharedPreferences pr=getSharedPreferences("LifeCalPrefs",MODE_PRIVATE);
                      String mode=pr.getString("mode","year");
                      long sMs=pr.getLong("startMillis",0),eMs=pr.getLong("endMillis",0);
                      Calendar t=Calendar.getInstance();
                      t.set(Calendar.HOUR_OF_DAY,0);t.set(Calendar.MINUTE,0);t.set(Calendar.SECOND,0);t.set(Calendar.MILLISECOND,0);
                      int total,dn;
                      if(mode.equals("year")){total=t.getActualMaximum(Calendar.DAY_OF_YEAR);dn=t.get(Calendar.DAY_OF_YEAR)-1;}
                      else if(sMs>0){
                          Calendar s=Calendar.getInstance(TimeZone.getTimeZone("UTC"));s.setTimeInMillis(sMs);s.set(Calendar.HOUR_OF_DAY,0);s.set(Calendar.MINUTE,0);s.set(Calendar.SECOND,0);s.set(Calendar.MILLISECOND,0);
                          Calendar e=Calendar.getInstance(TimeZone.getTimeZone("UTC"));e.setTimeInMillis(eMs>0?eMs:sMs+365L*86400000L);e.set(Calendar.HOUR_OF_DAY,0);e.set(Calendar.MINUTE,0);e.set(Calendar.SECOND,0);e.set(Calendar.MILLISECOND,0);
                          total=(int)((e.getTimeInMillis()-s.getTimeInMillis())/86400000L)+1;
                          dn=Math.max(0,Math.min((int)((t.getTimeInMillis()-s.getTimeInMillis())/86400000L),total));
                      }else{total=365;dn=0;}
                      int pct=total>0?(int)(dn*100.0/total):0;
                      float pad=W*0.05f;
                      Calendar now=Calendar.getInstance();
                      String time=String.format("%02d:%02d",now.get(Calendar.HOUR_OF_DAY),now.get(Calendar.MINUTE));
                      String[]mo={"Jan","Feb","Mar","Apr","May","Jun","Jul","Aug","Sep","Oct","Nov","Dec"};
                      String[]dy={"Sun","Mon","Tue","Wed","Thu","Fri","Sat"};
                      String ds=dy[now.get(Calendar.DAY_OF_WEEK)-1]+", "+now.get(Calendar.DAY_OF_MONTH)+" "+mo[now.get(Calendar.MONTH)];
                      clock.setTextSize(W*0.14f);c.drawText(time,W/2,H*0.12f,clock);
                      date.setTextSize(W*0.035f);c.drawText(ds.toUpperCase(),W/2,H*0.12f+W*0.045f,date);
                      float gTop=H*0.22f,gBot=H*0.88f,gW=W-pad*2;
                      int cols=20;float sp=gW/cols,r=sp*0.35f;
                      int rows=(int)Math.ceil((double)total/cols);
                      float sy=gTop+(gBot-gTop-rows*sp)/2f;
                      for(int i=0;i<total;i++){
                          float cx=pad+(i%cols)*sp+sp/2f,cy=sy+(i/cols)*sp+sp/2f;
                          if(i==dn){today.setShadowLayer(r*2,0,0,0x88ff4f6d);c.drawCircle(cx,cy,r*pulse,today);}
                          else if(i<dn){done.clearShadowLayer();c.drawCircle(cx,cy,r,done);}
                          else{left.clearShadowLayer();c.drawCircle(cx,cy,r,left);}
                      }
                      stats.setTextSize(W*0.042f);
                      c.drawText(dn+"d done  |  "+pct+"%  |  "+(total-dn)+"d left",W/2,gBot+(H-gBot)*0.35f,stats);
                      float by=gBot+(H-gBot)*0.68f,bh=H*0.005f;
                      c.drawRoundRect(new RectF(pad,by,W-pad,by+bh),bh/2,bh/2,barBg);
                      float fw=(W-pad*2)*pct/100f;
                      if(fw>0)c.drawRoundRect(new RectF(pad,by,pad+fw,by+bh),bh/2,bh/2,bar);
                  }
              }
          }
